# ПОЛУЧЕНИЕ ИЗМЕНЕНИЙ
git pull                    # Основная команда
git fetch                   # Безопаснее
git clone <url>            # Первый раз

# ПРОВЕРКА
git status                 # Текущее состояние
git log --oneline -5       # Последние 5 коммитов

# СОХРАНЕНИЕ СВОИХ ИЗМЕНЕНИЙ
git add .
git commit -m "Описание"
git push                   # Отправить на сервер
